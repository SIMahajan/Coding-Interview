5. Using Structures

struct Numbers
{
    int a;
    int b;
};

int main()
{
    // Initialize the structure
    struct Numbers nums = {5, 3};
    // Add the numbers
    int sum = nums.a + nums.b;
    // Print the result
    printf("Sum: %d\n", sum);
    return 0;
}

6. Using Recursion

 //Recursive function to add two Numbers

int add(int a, int b)
{
    if (b == 0)
    {
        return a;
    }
    return add(a + 1, b - 1);
}

int main()
{
    int a = 5;
    int b = 3;
    int sum = add(a, b);
    printf("Sum: %d\n", sum);
    return 0;
}
