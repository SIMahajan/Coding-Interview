6. Using Recursion

 //Recursive function to add two Numbers

int add(int a, int b)
{
    if (b == 0)
    {
        return a;
    }
    return add(a + 1, b - 1);
}

int main()
{
    int a = 5;
    int b = 3;
    int sum = add(a, b);
    printf("Sum: %d\n", sum);
    return 0;
}

5)Swap Two Numbers
#include <stdio.h>

int main()
{
    int a = 5, b = 10, temp;
    printf("Before swapping: a = %d, b = %d\n", a, b);

    // Swapping values of a and  b
    temp = a;
    a = b;
    b = temp;

    printf("After swapping: a = %d, b = %d\n", a, b);
    return 0;
}
/*Before swapping: a = 5, b = 10
After swapping: a = 10, b = 5*/
